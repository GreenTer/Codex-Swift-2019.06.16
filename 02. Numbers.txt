************************************        # как устроены параметры переменных
let number  = 255                           - Int - это 8байт! 64бита
let hex     = 0xff                          - 0x это шестнадцатиричное (max 255; 0xff это 255)
let binary  = 0b1111_1111                   - в 1байте, 8 символов двоичный (бинарный)
let oct     = 0o12                          - 0o1, это 8, а 8+2 = 10 (восьмеричный)

let number : UInt8 = 255                    - UInt8 имеет размер одного байта, и означает, что число без знаковое! диапозон от 0 до 255
let number : Int8 = 255                     - Int8 это число со знаком! диапозон от -128 до +127
13.12e2                                     - e2 означает 10 во 2ой степени, тоесть умножается на 100, e4 это 10000 и т.д.
13.12e-2                                    - e-2 означает 10 во 2ой степени, тоесть разделить на 100, e4 это 10000 и т.д.
0x5p2                                       - p2 означает 2 в степени 2, тоесть умножается на 4 и т.д.
0x5p-2                                      - p-2 означает 2 в степени 2, тоесть разделить на 4 и т.д.


************************************        # цифровые параметры
Int                                         - цифры (округляет до целого числа, в меньшую сторону всегда)
Float                                       - число, с плавающей запятой 32 битное, до 6 разрядов после запятой
Double                                      - число, с плавающей запятой 64 битное, до 15 значений после запятой

-----------------                           # опционал
(?) Еесли переменной задаём свойство с вопросительным знаком - это означает, что переменная может быть иногда nil!
(!) А если с восклицательным знаком - то nil'а не может быть!

Int                                         - объект не может быть nil
Int!                                        - объект не может быть nil (forced unwrapping)
Int?                                        - когда объект может быть nil, а может и не быть

-----------------                           # математика
(+) 1 + 2       // равно 3
(-) 5 - 3       // равно 2
(*) 2 * 3       // равно 6
(/) 10.0 / 2.5  // равно 4.0

var a = 1
a += 2 // теперь a равно 3

-----------------                           # примеры
let possibleString: String? = "An optional string."
let forcedString: String = possibleString!  // необходим восклицательный знак

let assumedString: String! = "An implicitly unwrapped optional string."
let implicitString: String = assumedString  // восклицательный знак не нужен

var surveyAnswer: String?                   // surveyAnswer автоматически установится в nil
var apples : Int! = nil
apples = 2
apples = apples + 5
